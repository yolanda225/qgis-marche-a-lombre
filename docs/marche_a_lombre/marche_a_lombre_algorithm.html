 74class MarcheALOmbreAlgorithm(QgsProcessingAlgorithm):
 75    """
 76    This is an example algorithm that takes a vector layer and
 77    creates a new identical one.
 78
 79    It is meant to be used as an example of how to create your own
 80    algorithms and explain methods and variables used to do it. An
 81    algorithm like this will be available in all elements, and there
 82    is not need for additional work.
 83
 84    All Processing algorithms should extend the QgsProcessingAlgorithm
 85    class.
 86    """
 87
 88    # Constants used to refer to parameters and outputs. They will be
 89    # used when calling the algorithm from another algorithm, or when
 90    # calling from the QGIS console.
 91
 92    OUTPUT = 'OUTPUT'
 93    INPUT = 'INPUT'
 94    DEPARTURE_TIME = 'DEPARTURE_TIME'
 95    HIKING_SPEED = 'HIKING_SPEED'
 96    PICNIC_POINT = 'PICNIC_POINT'
 97    PICNIC_DURATION = 'PICNIC_DURATION'
 98    REVERSE_DIRECTION = 'REVERSE_DIRECTION'
 99    BUFFER_MODE = 'BUFFER_MODE'
100    OUTPUT_POINTS = 'OUTPUT_POINTS'
101    LOW_RES_MNS = 'LOW_RES_MNS'
102    OUTPUT_CSV = 'OUTPUT_CSV'

104    def initAlgorithm(self, config):
105        """
106        Here we define the inputs and output of the algorithm, along
107        with some other properties.
108        """
109
110        # We add the input vector features source. It can have any kind of
111        # geometry.
112        self.addParameter(
113            QgsProcessingParameterFeatureSource(
114                self.INPUT,
115                self.tr('Input layer - tracks'),
116                [QgsProcessing.TypeVectorAnyGeometry]
117            )
118        )
119
120        self.addParameter(
121            QgsProcessingParameterDateTime(
122                self.DEPARTURE_TIME,
123                self.tr('Departure Date and Time (Local Time)'),
124                type=QgsProcessingParameterDateTime.DateTime  # Allows selecting both Date and Time
125            )
126        )
127
128        self.addParameter(
129            QgsProcessingParameterNumber(
130                self.HIKING_SPEED,
131                self.tr('Average Hiking Speed (km/h)'),
132                type=QgsProcessingParameterNumber.Double,
133                defaultValue=5.0,  # A standard hiking speed
134                minValue=0.0 
135            )
136        )
137
138        self.addParameter(
139            QgsProcessingParameterPoint(
140                self.PICNIC_POINT,
141                self.tr('Picnic Break Location (1h stop)'),
142                optional=True  # Optional: user might not want a break
143            )
144        )
145
146        self.addParameter(
147            QgsProcessingParameterNumber(
148                self.PICNIC_DURATION,
149                self.tr('Picnic Duration (minutes)'),
150                type=QgsProcessingParameterNumber.Integer,
151                defaultValue=60,
152                minValue=0,
153                optional=True
154            )
155        )
156
157        self.addParameter(
158            QgsProcessingParameterBoolean(
159                self.REVERSE_DIRECTION,
160                self.tr('Reverse Trail Direction (Finish to Start)'),
161                defaultValue=False
162            )
163        )
164
165        self.addParameter(
166            QgsProcessingParameterBoolean(
167                self.BUFFER_MODE,
168                self.tr('Calculate with Buffer (Center, Left 5m, Right 5m)'),
169                defaultValue=False
170            )
171        )
172
173
174        # We add a feature sink in which to store our processed features (this
175        # usually takes the form of a newly created vector layer when the
176        # algorithm is run in QGIS).
177        # self.addParameter(
178        #     QgsProcessingParameterFeatureSink(
179        #         self.OUTPUT,
180        #         self.tr('Output layer')
181        #     )
182        # )
183        self.addParameter(
184            QgsProcessingParameterRasterDestination(
185                self.OUTPUT,
186                self.tr('MNS')
187            )
188        )
189
190        self.addParameter(
191            QgsProcessingParameterFeatureSink(
192                self.OUTPUT_POINTS,
193                self.tr('Densified Trail Points'),
194                type=QgsProcessing.TypeVectorPoint
195            )
196        )
197
198        self.addParameter(
199            QgsProcessingParameterRasterDestination(
200                self.LOW_RES_MNS,
201                self.tr('Low Resolution MNS (Horizon)')
202            )
203        )
204
205        self.addParameter(
206            QgsProcessingParameterFileDestination(
207                self.OUTPUT_CSV,
208                self.tr('Statistics Table'),
209                fileFilter='CSV files (*.csv)'
210            )
211        )

213    def processAlgorithm(self, parameters, context, feedback):
214        """
215        Here is where the processing itself takes place.
216        """
217
218        # Retrieve the feature source and sink. The 'dest_id' variable is used
219        # to uniquely identify the feature sink, and must be included in the
220        # dictionary returned by the processAlgorithm function.
221        source = self.parameterAsSource(parameters, self.INPUT, context)
222        departure_dt = self.parameterAsDateTime(parameters, self.DEPARTURE_TIME, context)
223        speed = self.parameterAsDouble(parameters, self.HIKING_SPEED, context)
224        picnic_point = self.parameterAsPoint(parameters, self.PICNIC_POINT, context)
225        picnic_duration = self.parameterAsDouble(parameters, self.PICNIC_DURATION, context)
226        picnic_point_crs = self.parameterAsPointCrs(parameters, self.PICNIC_POINT, context)
227        reverse_direction = self.parameterAsBool(parameters, self.REVERSE_DIRECTION, context)
228        buffer_mode = self.parameterAsBool(parameters, self.BUFFER_MODE, context)
229        csv_path = self.parameterAsFileOutput(parameters, self.OUTPUT_CSV, context)
230
231        if not picnic_point or (picnic_point.x() == 0.0 and picnic_point.y() == 0.0):
232            picnic_point_crs = None
233
234        # Compute the number of steps to display within the progress bar and
235        # get features from source
236        total = 100.0 / source.featureCount() if source.featureCount() else 0
237        features = source.getFeatures()
238
239        ########################## TRAIL PROCESSING #########################
240        feedback.pushInfo("Processing trail...")
241        source_crs = source.sourceCrs()
242        if not source_crs.isValid():
243            raise QgsProcessingException(
244                "Input layer has no valid CRS. Please define the layer CRS before running this algorithm."
245            )
246
247        trail = Trail(
248            max_sep=10,
249            speed=speed, 
250            source_crs=source_crs,
251            transform_context=context.transformContext(),
252            feedback=feedback
253        )
254        trail.process_trail(source_tracks=source, 
255                            start_time=departure_dt, 
256                            break_point=picnic_point, 
257                            picnic_duration=picnic_duration,
258                            reverse=reverse_direction,
259                            buffer=buffer_mode,
260                            project_crs=picnic_point_crs)
261
262        ########################## MNT DOWNLOAD (For Z Values) ##################
263        # Generate a temporary file path for the MNT
264        mnt_path = QgsProcessingUtils.generateTempFilename('mnt_elevation.tif')
265        target_crs = trail.target_crs
266        downloader = MNSDownloader(
267            crs=target_crs, 
268            transform_context=context.transformContext(), 
269            feedback=feedback)
270        target_resolution = 0.5
271        # Download MNT (mns=False)
272        success_mnt = downloader.read_tif(
273            extent=trail.extent,
274            resolution=target_resolution*2, # less variation in trail elevation so lower resolution necessary
275            output_path=mnt_path,
276            input_crs=target_crs,
277            is_mns=False  
278        )
279
280        if not success_mnt:
281            raise QgsProcessingException("Failed to download MNT data.")
282        feedback.pushInfo(f"Downloading MNT for trail elevation data...")
283        
284        # Integrate MNT into Trail (Sample Z values)
285        feedback.pushInfo("Sampling elevation for trail points...")
286        trail.sample_elevation(mnt_path)
287
288        ########################## MNS DOWNLOAD (for Shade) ############################
289        output_path = self.parameterAsOutputLayer(parameters, self.OUTPUT, context)
290        low_res_path = self.parameterAsOutputLayer(parameters, self.LOW_RES_MNS, context)
291        
292        downloader = MNSDownloader(
293            crs=target_crs,
294            transform_context=context.transformContext(), 
295            feedback=feedback)
296
297        downloader.download_dual_quality_mns(
298            trail_extent=trail.extent,
299            high_res_path=output_path,
300            low_res_path=low_res_path,
301            trail_lat=trail.center_lat,
302            input_crs=target_crs,
303            high_res=target_resolution, # meters per pixel
304        )
305        
306        # Open MNS raster with GDAL
307        ds = gdal.Open(output_path)
308        if ds is None:
309            raise QgsProcessingException("Could not open downloaded MNS.")
310            
311        mns_band = ds.GetRasterBand(1)
312        mns_array = mns_band.ReadAsArray() # Returns numpy array
313        geo_transform = ds.GetGeoTransform()
314            
315        ds = None # Close dataset
316
317        ########################## CALCULATE SHADOWS ############################
318        feedback.pushInfo("Calculating shadows...")
319 
320        calculator = ShadowCalculator(
321            high_res_path=output_path,
322            low_res_path=low_res_path
323        )
324        
325        shadow_results = calculator.calculate_shadows(
326            trail_points=trail.trail_points,
327            max_dist_m=20000
328        )
329
330        ########################## CALCULATE STATISTICS ############################
331        total_points = len(shadow_results)
332        shady_points = sum(shadow_results)
333        sunny_points = total_points - shady_points
334        
335        percent_shady = (shady_points / total_points * 100) if total_points > 0 else 0.0
336        percent_sunny = (sunny_points / total_points * 100) if total_points > 0 else 0.0
337
338        feedback.pushInfo(f"Statistics: {percent_shady:.1f}% Shady, {percent_sunny:.1f}% Sunny")
339
340        if csv_path:
341            try:
342                with open(csv_path, 'w') as f:
343                    # Header
344                    f.write("Duration (min), % Shady, % Sunny, Time spent in Shadow (min), Time spent in Sun (min), Shady Points, Sunny Points, Total Points\n")
345                    
346                    duration_min = 0
347                    if trail.trail_points:
348                        start = trail.trail_points[0].datetime
349                        end = trail.trail_points[-1].datetime
350                        duration_min = start.secsTo(end) / 60
351                    time_shady = int(duration_min*percent_shady/100)
352                    time_sunny = int(duration_min*percent_sunny/100)
353
354                    f.write(f"{duration_min:.1f},{percent_shady:.2f},{percent_sunny:.2f},{time_shady},{time_sunny},{shady_points},{sunny_points},{total_points}\n")
355            except Exception as e:
356                feedback.reportError(f"Failed to write CSV: {e}")
357        
358        ########################## WRITE OUTPUT POINTS ##########################
359        # Define attribute table columns
360        fields = QgsFields()
361        fields.append(QgsField("id", QVariant.Int))
362        fields.append(QgsField("status", QVariant.String))     # Sunny / Shady
363        fields.append(QgsField("is_shadow", QVariant.Int))     # 0 / 1
364        fields.append(QgsField("x_proj", QVariant.Double))      # Lambert-93 X
365        fields.append(QgsField("y_proj", QVariant.Double))      # Lambert-93 Y
366        fields.append(QgsField("z_mnt", QVariant.Double))      # Altitude
367        fields.append(QgsField("latitude", QVariant.Double))   # WGS84 Lat
368        fields.append(QgsField("longitude", QVariant.Double))  # WGS84 Lon
369        fields.append(QgsField("arrival_time", QVariant.DateTime)) # Time
370        fields.append(QgsField("elevation_deg", QVariant.Double)) # Sun elevation angle
371        fields.append(QgsField("azimuth_deg", QVariant.Double)) # Sun azimtuh angle
372        fields.append(QgsField("course", QVariant.Double))
373
374        (point_sink, point_dest_id) = self.parameterAsSink(
375            parameters, 
376            self.OUTPUT_POINTS, 
377            context, 
378            fields, 
379            QgsWkbTypes.PointZ,
380            QgsCoordinateReferenceSystem(target_crs)
381        )
382        self.point_dest_id = point_dest_id
383
384        if point_sink is None:
385            raise QgsProcessingException("Could not create point sink")
386        
387        feedback.pushInfo("Writing results...")
388        for i, tp in enumerate(trail.trail_points):
389            if feedback.isCanceled(): break
390
391            # Calculate direction to next point for visualization
392            course = 0.0
393            if i < len(trail.trail_points) - 1:
394                next_tp = trail.trail_points[i+1]
395                course = QgsPoint(tp.x, tp.y).azimuth(QgsPoint(next_tp.x, next_tp.y))
396            elif i > 0:
397                course = prev_course
398            prev_course = course
399
400            feat = QgsFeature(fields)
401            geom = QgsGeometry.fromPoint(QgsPoint(tp.x, tp.y, tp.z))
402            feat.setGeometry(geom)
403            
404            is_shadow = shadow_results[i]
405            status_str = "Shadow" if is_shadow == 1 else "Sun"
406            
407            feat.setAttributes([
408                i,
409                status_str,
410                int(is_shadow),
411                tp.x,
412                tp.y,
413                tp.z,
414                tp.lat,
415                tp.lon,
416                tp.datetime,
417                math.degrees(tp.solar_pos[0]),
418                math.degrees(tp.solar_pos[1]),
419                course
420            ])
421            point_sink.addFeature(feat, QgsFeatureSink.FastInsert)
422            feedback.setProgress(int(i * total))
423
424        # Return the results of the algorithm. In this case our only result is
425        # the feature sink which contains the processed features, but some
426        # algorithms may return multiple feature sinks, calculated numeric
427        # statistics, etc. These should all be included in the returned
428        # dictionary, with keys matching the feature corresponding parameter
429        # or output names.
430        self.results = {
431            self.OUTPUT: output_path,
432            self.OUTPUT_POINTS: point_dest_id,
433            self.LOW_RES_MNS: low_res_path,
434            self.OUTPUT_CSV: csv_path,
435            'detected_crs': target_crs
436        }
437        return self.results
438    
439    def postProcessAlgorithm(self, context, feedback):
440        # Set project CRS to match the detected region
441        detected_crs_str = self.results.get('detected_crs')
442        if detected_crs_str:
443            detected_crs = QgsCoordinateReferenceSystem(detected_crs_str)
444            project = context.project()
445            
446            if detected_crs.isValid() and project.crs().authid() != detected_crs.authid():
447                feedback.pushInfo(f"Changing project CRS from {project.crs().authid()} to {detected_crs.authid()} to match detected region")
448                project.setCrs(detected_crs)
449        # style the output trail depending on Sun/Shadow
450        layer = QgsProcessingUtils.mapLayerFromString(self.point_dest_id, context)
451        
452        if layer:
453            categories = []
454            styles = [
455                (0, "Sun", "gold"), 
456                (1, "Shadow", "#1f78b4") 
457            ]
458
459            for val, label, color in styles:
460                sym_layer = QgsSimpleMarkerSymbolLayer.create({
461                    'name': 'filled_arrowhead', 
462                    'color': color,
463                    'size': '2.5',
464                })
465
466                # Direction of arrow based on course field
467                prop_angle = QgsProperty.fromExpression('"course" - 90')
468                sym_layer.setDataDefinedProperty(QgsSymbolLayer.PropertyAngle, prop_angle)
469
470                sym = QgsSymbol.defaultSymbol(layer.geometryType())
471                sym.changeSymbolLayer(0, sym_layer)
472                categories.append(QgsRendererCategory(val, sym, label, True))
473
474            renderer = QgsCategorizedSymbolRenderer("is_shadow", categories)
475            layer.setRenderer(renderer)
476            layer.triggerRepaint()
477
478        csv_path = self.results.get(self.OUTPUT_CSV)
479        if csv_path and os.path.exists(csv_path):
480            # Create delimited text layer from csv
481            path = csv_path.replace('\\', '/')
482            uri = f"file:///{path}?type=csv&watchFile=no"
483            
484            vlayer = QgsVectorLayer(uri, "Shadow Statistics", "delimitedtext")  
485            if vlayer.isValid():
486                context.project().addMapLayer(vlayer)
487        return self.results
